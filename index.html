Question 4 : Even Odd

Answer:

const findEvenOdd = (N, Arr) => 
{
    let sumOfEven=0;
    let sumOfOdd=0;
    for(let i=0;i < N;i++){
      if(Arr[i]%2===0){
        sumOfEven=sumOfEven+Arr[i];
      }
      else{
        sumOfOdd=sumOfOdd+Arr[i];
      }
    }
    let B=[];
    B.push(sumOfEven);
    B.push(sumOfOdd);
    return B;
};

Question 5 : Find whether the number is present or not

Answer:

const Find_Num = (array,N,M) => 
{
  let count=0;
  for(let i=0;i < N;i++){
    if(array[i]==M){
      count++;
    }
  }
  if(count>0){
    return "YES";
  }
  else{
    return "NO";
  }
};

Question 9 : Unique Color Shirt

Answer:

function Unique_Shirts (arr,N) {
  let uniqueElement=0;
  for(let j=0;j < arr.length;j++){
    let count=0;
    let key=arr[j];
    for(i=0;i < N;i++){
      if(key==arr[i]){
        count++;
      }
    }
      if(count==1){
        uniqueElement++;
      }
  }
 return uniqueElement;
}
